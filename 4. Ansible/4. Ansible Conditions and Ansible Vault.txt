inside ansible
# vi condition.yml  
--- # My playbook for condition
- hosts: demo
  remote_user: ansible
  become: yes
  connection: ssh
  
  tasks:
    - name: Install apache on Debian family
      ansible.builtin.apt:
        name: apache2
        state: present
      when: ansible_os_family == "Debian"

    - name: Install apache on RedHat family
      ansible.builtin.dnf:
        name: httpd
        state: present
      when: ansible_os_family == "RedHat"

For execute:
# ansible-playbook condition.yml
