###--- 6. Jenkins
helm repo add jenkins https://charts.jenkins.io --force-update
helm repo update

kubectl create namespace jenkins
helm install jenkins jenkins/jenkins -n jenkins
kubectl get pods -n jenkins

kubectl get svc -n jenkins
kubectl edit deployment jenkins-0 -n jenkins
kubectl edit statefulset jenkins -n jenkins

kubectl patch svc jenkins -n jenkins -p '{"spec": {"type": "NodePort"}}'

Get Admin Password:
kubectl exec --namespace jenkins -it svc/jenkins -c jenkins -- cat /run/secrets/additional/chart-admin-password && echo
