# ll === Display directory contents (list)
# ls === Display directory contents (list)

# ls -Z === Security admin
# ls home/ === Have anything file on home?

# cd === Go to login location
# cd name === Forword one step
# cd name/name === Forword two step
# cd .. === Backword one step
# cd ../.. === Backword two step
# cd ../../media === Backword two step & Forword one step
# cd - === Go previous location

# echo Today  is good day === Print
# touch mytest === create file
# echo Today  is friday > mytest === Text write on mytest file / Or overwrite data
# cat mytest === Open file
# echo I am naiim >> mytest === Add new text on mytest file
# echo -e "apple\nred\norange\nbanana" >> testfile
# touch - myfile textfile amnifile === 3 text file created
# rm -f myfile textfile === Remove 2 file
# touch file{10..15} === Multiple file created
# touch Ban/Dhaka/File{1..5} === Created file on folder
# rm -f Ban/Dhaka/File{1..3} === Removed file on folder
# rm -rf Ban/ === Removed all on Ban 

# head file1 === Top data show (like cat)
# head -n 2 file1 === Top 2 line file data show 
# tail file1 === Bottom data show (like cat)
# tail -n 2 file1 === Bottom 2 line file data show 
# sort mytest === Data sort (asc)
# sort -r mytest === Data reverse (desc)
# grep "P" mytest === Find out all P
# grep "P" mytest | grep "Po" === Find out all P and PO
# less mytest === For many text 
# more mytest === For many text (%)

# mkdir my_folder === Folder create
# mkdir folder1 folder2 folder3 === Multiple folder create

wc file.txt
wc -l file.txt → শুধু lines count করবে
wc -w file.txt → শুধু words count করবে
wc -c file.txt → শুধু bytes (characters) count করবে
wc -m file.txt → শুধু characters count করবে (multi-byte aware)
wc -L file.txt → longest line এর length দেখাবে
