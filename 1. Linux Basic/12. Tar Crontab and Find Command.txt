Tar/Zip folder: [c → create, v → verbose, f → file, z → gzip compression]

# ls -l data.tar
# tar -tvf data.tar === show with details

Search/filter:
file = f
directory = d
# find / -type f -iname Passwd === file search
# find / -type d -iname naiim === directory search
# find / -iname naiim === show all

-amin = files accessed before minutes
-cmin = files created before minutes
-mmin = files modified before minutes
-atime = days
-ctime = days
-mtime = days
# find / -mmin 5 === modify 5 min before
# find / -cmin 5 === modify 5 min before
