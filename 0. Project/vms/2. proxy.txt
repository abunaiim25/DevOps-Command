‡¶§‡¶æ‡¶∞‡¶™‡¶∞ ‡¶≤‡¶ø‡¶ñ‡ßá ‡¶¶‡ßá ‚¨áÔ∏è
(‡¶è‡¶á‡¶ü‡¶æ ‡¶Ü‡¶Æ‡¶ø SSL ‡¶∏‡¶π version, self-signed certificate setup ‡¶∏‡¶π)

# Define upstream (backend) server group
upstream test_vms_backend {
    server 192.168.10.203:8080;  # Tomcat or backend app
    keepalive 32;
}

# HTTP -> HTTPS redirect
server {
    listen 80;
    server_name test-vms.oss.net.bd;
    return 301 https://$host$request_uri;
}

# HTTPS server
server {
    listen 443 ssl;
    server_name test-vms.oss.net.bd;

    ssl_certificate /etc/ssl/certs/test-vms.crt;
    ssl_certificate_key /etc/ssl/private/test-vms.key;

    location / {
        proxy_pass http://test_vms_backend;   # use upstream name here
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
