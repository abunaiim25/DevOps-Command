Docker website: https://docs.docker.com/engine/install/rhel/

Update & install Docker/Docker Compose
# sudo dnf update -y

Install using the rpm repository:
# sudo dnf -y install dnf-plugins-core
  sudo dnf config-manager --add-repo https://download.docker.com/linux/rhel/docker-ce.repo

Install Docker Engine:
# sudo dnf install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin

# which docker === docker have or not
  docker -v
  
Start Docker Engine:
# sudo systemctl enable --now docker
  sudo systemctl status docker 
  or
  service docker start === docker start

# docker start or not (status):
  service docker status  
  docker info
  
-----------------------------Basic Commands----------------------------------
# docker images
  docker search ubuntu
  docker pull ubuntu
  docker run -it ubuntu /bin/bash
  or
  docker run -it --name amirayhan ubuntu /bin/bash
  
root@35a118334730:/#
# ls
  cat /etc/os-release === OS check
  ubuntu -v
  exit

# docker images
  docker ps
  docker pa -a
  
Start container:
# docker start 35a118334730
  docker start epic_shockley
  
To go inside container:
# docker attach amirayhan
  
Start container:
# docker stop 35a118334730
  docker stop amirayhan
  
Stop container remove:
# docker rm 35a118334730
  docker rm amirayhan

Stop all running containers:
# sudo docker stop $(sudo docker ps -a -q)  

Deleted all stopped containers:
# sudo docker rm $(sudo docker ps -a -q) 

Deleted all images:
# sudo docker rmi -f $(sudo docker images -q)
