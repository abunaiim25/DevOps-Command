Backup:
podman cp tomcat-https:/opt/tomcat /home/naiim/tomcat-https-backup

container to Image Commit:
podman commit tomcat-https tomcat-https-backup

Image to container run:
podman run -d --name tomcat-https-restored \
  -p 8443:8443 \
  -v ./tomcat-conf/server.xml:/opt/tomcat/conf/server.xml:Z \
  -v /root/keystore.jks:/opt/tomcat/conf/keystore.jks:Z \
  -v /root/logs:/opt/tomcat/logs:Z \
  localhost/tomcat-https-backup \
  /bin/sh -c "/opt/tomcat/bin/catalina.sh run >> /opt/tomcat/logs/stdout.log 2>> /opt/tomcat/logs/stderr.log"
