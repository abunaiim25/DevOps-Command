kubectl get secret
kubectl describe secret mysecret


###--- Secrect for pod
vi secrectpod.yml

apiVersion: v1
kind: Pod
metadata:
  name: myvolsecret
spec:
  containers:
    - name: c1
      image: centos:7
      command: ["/bin/bash", "-c", "while true; do echo Technical-guftgu; sleep 5 ; done"]
      volumeMounts:
        - name: testsecret
          mountPath: "/tmp/mysecrets"   # Secret ফাইলগুলো এখানে read-only হিসেবে mount হবে
  volumes:
    - name: testsecret
      secret:
        secretName: mysecret   # Secret এর নাম


Apply and check:
kubectl apply -f secrectpod.yml
kubectl get pods 
kubectl describe pod myvolsecret
kubectl delete pod myvolsecret

Inside container:
kubectl exec -it myvolsecret -- /bin/bash
cd /tmp/mysecrets
ls
