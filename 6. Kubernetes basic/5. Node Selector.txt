1. Label node:
kubectl get nodes
kubectl label nodes minikube/worker-node-1 hardware=t2-medium

2. Verify
kubectl get nodes --show-labels

3. nodeSelector on Pod YAML
vi pod3.yaml  

apiVersion: v1                  # Kubernetes API version
kind: Pod                       # This resource is a Pod
metadata:
  name: testpod                 # Pod name
  annotations:
    description: I am abu Naiim  # Optional description
  labels:
    app: testpod                # Label to identify the app
    owner: naiim                # Label to identify owner
    myenv: dev                  # Label to identify environment

spec:
  containers:
    - name: ubuntu-container     # First container name
      image: ubuntu              # Image to use for the container
      command: ["/bin/bash", "-c", "while true; do echo Hello-Naiim; sleep 5; done"]  # Command to run continuously
      env:
        - name: MY_NAME
          value: "Abu Naiim"    # Environment variable inside container
      ports:
        - containerPort: 8080   # Port exposed by this container

    - name: httpd-container      # Second container name
      image: httpd               # Apache HTTPD image
      command: ["/bin/bash", "-c", "while true; do echo Apache httpd; sleep 5; done"]  # Keep container running
      env:
        - name: MY_NAME
          value: "Abu Naiim"    # Environment variable inside container
      ports:
        - containerPort: 80     # Port exposed by HTTPD container

  restartPolicy: Never           # Pod will not restart if container exits

  nodeSelector:                  # Node selector for scheduling
    hardware: t2-medium          # Must match node label


4. Apply and verify
kubectl apply -f pod3.yaml 
kubectl describe pod testpod 
kubectl get pods
